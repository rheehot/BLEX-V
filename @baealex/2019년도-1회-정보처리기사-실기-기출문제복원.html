  • LM : 문제에서 주어진 수의 개수

2. 다음 소스코드의 출력결과를 작성하시오. 단, println 함수를 사용했음을 고려하시오.

코드

class SuperClass {
    void paint() {
        draw();
    }
    void draw() {
        system.out.println("Super Object");
    }
}

class SubClass extends SuperClass {
    void paint() {
        super.paint();
        super.draw();
    }
    void draw() {
        system.out.println("Sub Object");
    }
}

public class SampleProgram {
    public static void main(String[] args) {
        SuperClass ex = new SubClass();
        ex.paint();
    }
}

코드

#include <stdio.h>

int main(void)
{
    int input = 0;
    int sum = 0;

    scanf("%d", &input);

    while(1)
    {
        if([  1  ] == 0)
        {
            break;
        }
        sum = sum + input%[  2  ];
        input = input/[  2  ];
    }

    printf("%d\n", sum);
    return 0;
}
